# BaseX

## Install
`npm i --save @dwlib/basex`

## Usage
```javascript
// CJS
const BaseX = require('@dwlib/basex');
const BaseXPrototype = require('@dwlib/basex/prototype');
const BaseXPrototypeDecode = require('@dwlib/basex/prototype/decode');
const BaseXPrototypeDecodeBigInt = require('@dwlib/basex/prototype/decodeBigInt');
const BaseXPrototypeDecodeInt = require('@dwlib/basex/prototype/decodeInt');
const BaseXPrototypeDecodeInto = require('@dwlib/basex/prototype/decodeInto');
const BaseXPrototypeEncode = require('@dwlib/basex/prototype/encode');
const BaseXPrototypeEncodeBigInt = require('@dwlib/basex/prototype/encodeBigInt');
const BaseXPrototypeEncodeInt = require('@dwlib/basex/prototype/encodeInt');
const BaseXPrototypeIsValid = require('@dwlib/basex/prototype/isValid');
const BaseXPrototypeNormalize = require('@dwlib/basex/prototype/normalize');
const BaseXPrototypeValidate = require('@dwlib/basex/prototype/validate');
// ESM
import BaseX from '@dwlib/basex';
import BaseXPrototype from '@dwlib/basex/prototype';
import BaseXPrototypeDecode from '@dwlib/basex/prototype/decode';
import BaseXPrototypeDecodeBigInt from '@dwlib/basex/prototype/decodeBigInt';
import BaseXPrototypeDecodeInt from '@dwlib/basex/prototype/decodeInt';
import BaseXDecodeInto from '@dwlib/basex/prototype/decodeInto';
import BaseXPrototypeEncode from '@dwlib/basex/prototype/encode';
import BaseXPrototypeEncodeBigInt from '@dwlib/basex/prototype/encodeBigInt';
import BaseXPrototypePrototypeEncodeInt from '@dwlib/basex/prototype/encodeInt';
import BaseXPrototypeIsValid from '@dwlib/basex/prototype/isValid';
import BaseXPrototypeNormalize from '@dwlib/basex/prototype/normalize';
import BaseXPrototypeValidate from '@dwlib/basex/prototype/validate';
```
